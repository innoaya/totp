function generateHOTP(secret, counter) {
  // 1. Counter to 8-byte hex
  let hex = counter.toString(16).padStart(16,'0');
  let msg = CryptoJS.enc.Hex.parse(hex);
  
  // 2. HMAC-SHA1
  let hmac = CryptoJS.HmacSHA1(msg, secret);
  let hash = hmac.toString(CryptoJS.enc.Hex);
  
  // 3. Dynamic Truncation
  let offset = parseInt(hash.slice(-1), 16);
  let pHash = hash.slice(offset*2, offset*2+8);
  let bin = parseInt(pHash, 16) & 0x7fffffff;
  
  // 4. Extract 6 digits
  let otp = bin % 1000000;
  return otp.toString().padStart(6,'0');
}

function generateTOTP(secret) {
  // 1. Get Unix Time in seconds
  let now = Math.floor(Date.now() / 1000);
  
  // 2. Create Counter (Time / 30)
  // Which 30s slot are we in?
  let counter = Math.floor(now / 30);
  
  // 3. Use HOTP with Time Counter
  // TOTP is just HOTP using Time
  return generateHOTP(secret, counter);
}
